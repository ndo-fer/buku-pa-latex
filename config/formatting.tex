\makeatletter
\def\cleardoublepage{%
  \clearpage
  \if@twoside
  \ifodd\c@page\else
  \hbox{}
  \vspace*{\fill}
  \begin{center}
    \textit{Halaman ini sengaja dikosongkan}
  \end{center}
  \vspace{\fill}
  \thispagestyle{fancy}
  \newpage
  \if@twocolumn\hbox{}\newpage\fi
  \fi
  \fi
}
\makeatother
